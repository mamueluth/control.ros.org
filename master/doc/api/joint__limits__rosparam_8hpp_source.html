36 template <typename ParameterT>
37 auto auto_declare(
38  const rclcpp::node_interfaces::NodeParametersInterface::SharedPtr & param_itf,
39  const std::string & name, const ParameterT & default_value)
40 {
41  if (!param_itf->has_parameter(name))
42  {
43  auto param_default_value = rclcpp::ParameterValue(default_value);
44  param_itf->declare_parameter(name, param_default_value);
45  }
46  return param_itf->get_parameter(name).get_value<ParameterT>();
47 }

216 inline bool get_joint_limits(
217  const std::string & joint_name,
218  const rclcpp::node_interfaces::NodeParametersInterface::SharedPtr & param_itf,
219  const rclcpp::node_interfaces::NodeLoggingInterface::SharedPtr & logging_itf,
220  JointLimits & limits)
221 {
222  const std::string param_base_name = "joint_limits." + joint_name;
223  try
224  {
225  if (
226  !param_itf->has_parameter(param_base_name + ".has_position_limits") &&
227  !param_itf->has_parameter(param_base_name + ".min_position") &&
228  !param_itf->has_parameter(param_base_name + ".max_position") &&
229  !param_itf->has_parameter(param_base_name + ".has_velocity_limits") &&
230  !param_itf->has_parameter(param_base_name + ".min_velocity") &&
231  !param_itf->has_parameter(param_base_name + ".max_velocity") &&
232  !param_itf->has_parameter(param_base_name + ".has_acceleration_limits") &&
233  !param_itf->has_parameter(param_base_name + ".max_acceleration") &&
234  !param_itf->has_parameter(param_base_name + ".has_jerk_limits") &&
235  !param_itf->has_parameter(param_base_name + ".max_jerk") &&
236  !param_itf->has_parameter(param_base_name + ".has_effort_limits") &&
237  !param_itf->has_parameter(param_base_name + ".max_effort") &&
238  !param_itf->has_parameter(param_base_name + ".angle_wraparound") &&
239  !param_itf->has_parameter(param_base_name + ".has_soft_limits") &&
240  !param_itf->has_parameter(param_base_name + ".k_position") &&
241  !param_itf->has_parameter(param_base_name + ".k_velocity") &&
242  !param_itf->has_parameter(param_base_name + ".soft_lower_limit") &&
243  !param_itf->has_parameter(param_base_name + ".soft_upper_limit"))
244  {
245  RCLCPP_ERROR(
246  logging_itf->get_logger(),
247  "No joint limits specification found for joint '%s' in the parameter server "
248  "(param name: %s).",
249  joint_name.c_str(), param_base_name.c_str());
250  return false;
251  }
252  }
253  catch (const std::exception & ex)
254  {
255  RCLCPP_ERROR(logging_itf->get_logger(), "%s", ex.what());
256  return false;
257  }
258 
259  // Position limits
260  if (param_itf->has_parameter(param_base_name + ".has_position_limits"))
261  {
262  limits.has_position_limits =
263  param_itf->get_parameter(param_base_name + ".has_position_limits").as_bool();
264  if (
265  limits.has_position_limits && param_itf->has_parameter(param_base_name + ".min_position") &&
266  param_itf->has_parameter(param_base_name + ".max_position"))
267  {
268  limits.min_position = param_itf->get_parameter(param_base_name + ".min_position").as_double();
269  limits.max_position = param_itf->get_parameter(param_base_name + ".max_position").as_double();
270  }
271  else
272  {
273  limits.has_position_limits = false;
274  }
275 
276  if (
277  !limits.has_position_limits &&
278  param_itf->has_parameter(param_base_name + ".angle_wraparound"))
279  {
280  limits.angle_wraparound =
281  param_itf->get_parameter(param_base_name + ".angle_wraparound").as_bool();
282  }
283  }
284 
285  // Velocity limits
286  if (param_itf->has_parameter(param_base_name + ".has_velocity_limits"))
287  {
288  limits.has_velocity_limits =
289  param_itf->get_parameter(param_base_name + ".has_velocity_limits").as_bool();
290  if (limits.has_velocity_limits && param_itf->has_parameter(param_base_name + ".max_velocity"))
291  {
292  limits.max_velocity = param_itf->get_parameter(param_base_name + ".max_velocity").as_double();
293  }
294  else
295  {
296  limits.has_velocity_limits = false;
297  }
298  }
299 
300  // Acceleration limits

302  {
303  limits.has_acceleration_limits =
304  param_itf->get_parameter(param_base_name + ".has_acceleration_limits").as_bool();
305  if (
306  limits.has_acceleration_limits &&
307  param_itf->has_parameter(param_base_name + ".max_acceleration"))
308  {
309  limits.max_acceleration =

311  }
312  else

314  limits.has_acceleration_limits = false;
315  }
316  }
317 
318  // Jerk limits

320  {
321  limits.has_jerk_limits =
322  param_itf->get_parameter(param_base_name + ".has_jerk_limits").as_bool();
323  if (limits.has_jerk_limits && param_itf->has_parameter(param_base_name + ".max_jerk"))

325  limits.max_jerk = param_itf->get_parameter(param_base_name + ".max_jerk").as_double();
326  }

424 inline bool get_joint_limits(
425  const std::string & joint_name,
426  const rclcpp::node_interfaces::NodeParametersInterface::SharedPtr & param_itf,
427  const rclcpp::node_interfaces::NodeLoggingInterface::SharedPtr & logging_itf,
428  SoftJointLimits & soft_limits)
429 {
430  const std::string param_base_name = "joint_limits." + joint_name;
431  try
432  {
433  if (
434  !param_itf->has_parameter(param_base_name + ".has_soft_limits") &&
435  !param_itf->has_parameter(param_base_name + ".k_velocity") &&
436  !param_itf->has_parameter(param_base_name + ".k_position") &&
437  !param_itf->has_parameter(param_base_name + ".soft_lower_limit") &&
438  !param_itf->has_parameter(param_base_name + ".soft_upper_limit"))
439  {
440  RCLCPP_DEBUG(
441  logging_itf->get_logger(),
442  "No soft joint limits specification found for joint '%s' in the parameter server "
443  "(param name: %s).",
444  joint_name.c_str(), param_base_name.c_str());
445  return false;
446  }
447  }
448  catch (const std::exception & ex)
449  {
450  RCLCPP_ERROR(logging_itf->get_logger(), "%s", ex.what());
451  return false;
452  }
453 
454  // Override soft limits if complete specification is found
455  if (param_itf->has_parameter(param_base_name + ".has_soft_limits"))
456  {
457  if (
458  param_itf->get_parameter(param_base_name + ".has_soft_limits").as_bool() &&
459  param_itf->has_parameter(param_base_name + ".k_position") &&
460  param_itf->has_parameter(param_base_name + ".k_velocity") &&
461  param_itf->has_parameter(param_base_name + ".soft_lower_limit") &&
462  param_itf->has_parameter(param_base_name + ".soft_upper_limit"))
463  {
464  soft_limits.k_position =
465  param_itf->get_parameter(param_base_name + ".k_position").as_double();
466  soft_limits.k_velocity =
467  param_itf->get_parameter(param_base_name + ".k_velocity").as_double();
468  soft_limits.min_position =
469  param_itf->get_parameter(param_base_name + ".soft_lower_limit").as_double();
470  soft_limits.max_position =
471  param_itf->get_parameter(param_base_name + ".soft_upper_limit").as_double();
472  return true;
473  }
474  }
475 
476  return false;
477 }
