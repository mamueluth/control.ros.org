159  template <typename T>
160  std::shared_ptr<typename T::Response> call_service_and_wait(
161  rclcpp::Client<T> & client, std::shared_ptr<typename T::Request> request,
162  rclcpp::Executor & service_executor, bool update_controller_while_spinning = false)
163  {
164  EXPECT_TRUE(client.wait_for_service(std::chrono::milliseconds(500)));
165  auto result = client.async_send_request(request);
166  // Wait for the result.
167  if (update_controller_while_spinning)
168  {
169  while (service_executor.spin_until_future_complete(result, std::chrono::milliseconds(50)) !=
170  rclcpp::FutureReturnCode::SUCCESS)
171  {
172  cm_->update(rclcpp::Time(0), rclcpp::Duration::from_seconds(0.01));
173  }
174  }
175  else
176  {
177  EXPECT_EQ(
178  service_executor.spin_until_future_complete(result), rclcpp::FutureReturnCode::SUCCESS);
179  }
180  return result.get();
181  }
182 
183 protected:
184  rclcpp::TimerBase::SharedPtr update_timer_;
185  std::future<void> executor_spin_future_;
186 };
